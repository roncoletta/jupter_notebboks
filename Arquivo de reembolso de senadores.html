def gastos_categorizado(senador):
    nome_senador = df[df.SENADOR == senador]
    nome_senador = nome_senador.groupby('TIPO_DESPESA')[['SENADOR', 'VALOR_REEMBOLSADO']].sum()
    return nome_senador
